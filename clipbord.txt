const searchBtn = document.getElementById("search-btn");
const nav = document.getElementById("navigation");

async function getWeather() {

    if(!city){
        alert("Please enter your city.");
        return;
    }

    const apiKey = 50c72c73c56d68e2105dd2210c9b52df;
    const city = document.getElementById("city").value;

    const currentWeatherUrl = `https://api.openweathermap.org/data/2.5/weather?q=${city}&appid=${apiKey}`;
    const forecastUrl = `https://api.openweathermap.org/data/2.5/forecast?q=${city}&appid=${apiKey}`;

    // fetch current weather data
    try{
        const weatherResponse = await fetch(currentWeatherUrl);
        const weatherData = await weatherResponse.json();
        await displayWeather(weatherData);
    } catch(error) {
        console.error(`Error fetching current weather data: ${error.message}`);
        alert(`Error fetching current weather data please try again.`);
    }

    // fetch hourly forecast data
    try {
        const hourlyForecast = await fetch(forecastUrl);
        const hourlyForecastData = await hourlyForecast.json();
        await displayHourlyForecast(hourlyForecastData.list);
    } catch (error) {
        console.error(`Error fetching hourly forecast data: ${error.message}`);
        alert(`Error fetching hourly forecast data please try again.`);
    }

    weatherContainer.classList.remove('hidden');
    nav.classList.remove('h-screen');
}

function displayWeather(data){

    const weatherTemp = document.getElementById('weather-temp');
    const weatherIcon = document.getElementById('weather-icon');
    const weatherInfo = document.getElementById('weather-info');
    const date = document.getElementById('date');
    const location = document.getElementById('location');
    
    const hourlyForecast = document.getElementById('todays-highlights');
    
    const futureDaysForecast = document.getElementById('five-days-forecast');

    const airQualityIndex = document.getElementById('air-quality-index');
    const sunriseSunset = document.getElementById('sunrise-sunset');
    const humidity = document.getElementById('humidity');
    const airPressure = document.getElementById('pressure');
    const visibility = document.getElementById('visibility');
    const feelsLike = document.getElementById('feels-like');

    // Clear previous content.
    weatherTemp.innerHTML = '';
    weatherIcon.src = '';
    weatherInfo.innerHTML = '';
    date.innerHTML = '';
    location.innerHTML = '';
    
    futureForecast.innerHTML = '';
    hourlyForecast.innerHTML = '';
    airQualityIndex.innerHTML = '';
    sunriseSunset.innerHTML = '';
    humidity.innerHTML = '';
    airPressure.innerHTML = '';
    visibility.innerHTML = '';
    feelsLike.innerHTML = '';

    if(data.cod === '404') {
        weatherInfo.innerHTML = `<p>${data.message}</p>`;
    } else {
        const cityName = data.name;
        const temperature = Math.round(data.main.temp - 273.15); // Convert from kelvin to celsius
        const description = data.weather[0].description;
        const iconCode = data.weather[0].icon;
        const iconUrl = `https://openweathermap.org/img/wn/${iconCode}@4x.png`;

        const temperatureHTML = `
            <p>${temperature}&deg;C</p>
        `;
        const weatherHTML = `
            <p>${cityName}</p>
            <p>${description}</p>
        `;
    
        weatherTemp.innerHTML = temperatureHTML;
        weatherInfo.innerHTML = weatherHTML;
        weatherIcon.src = iconUrl;
        weatherIcon.alt = description;

        showImage();
    }
}

function displayHourlyForecast(hourlyData) {
    const hourlyForecast = document.getElementById('todays-highlights');
    const next24Hours = hourlyData.slice(0, 8);

    next24Hours.forEach(item => {
        const dateTime = new Date(item.dt * 1000);
        const hour = dateTime.getHours();
        const temperatureHTML = Math.round(item.main.temp - 273.15);
        const iconUrl = `https://openweathermap.org/img/wn/${iconCode}@4x.png`;

        const hourlyItemHtml = `
            <div class="hourly-item">
                <span>${hour}:00</span>
                <img src="${iconUrl}" alt="Hourly Weather Icon">
                <span>${temperature}&deg;C</span>
            </div>
        `;
        hourlyForecast.innerHTML += hourlyItemHtml;
    });
}

function showImage() {
    const weatherIcon = document.getElementById('weather-icon');
    weatherIcon.style.display = 'block';
}

searchBtn.addEventListener('click', () => {

   getWeather();
});
